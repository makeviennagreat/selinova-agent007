SQL Fix — Businesses Rating

      

      

        

          
 Problem: ERROR 22P02 — invalid input syntax for numeric: "4,4"

          

            Führe diesen SQL-Code im Supabase SQL Editor deines Projekts (jvmpojigydbkrzlabuor) aus, um die Tabelle zu reparieren.
          

          

            
-- SCHRITT 1: Spalte temporär auf TEXT ändern (damit "4,4" importiert werden kann)
ALTER TABLE businesses ALTER COLUMN rating TYPE TEXT;

-- SCHRITT 2: Jetzt deine CSV normal im Table Editor importieren

-- SCHRITT 3: Nach dem Import — Komma durch Punkt ersetzen und zurückkonvertieren
UPDATE businesses
  SET rating = REPLACE(rating, ',', '.')
  WHERE rating LIKE '%,%';

ALTER TABLE businesses
  ALTER COLUMN rating TYPE NUMERIC(3,1) USING rating::NUMERIC;

-- SCHRITT 4: RLS aktivieren (falls noch nicht aktiv)
ALTER TABLE businesses ENABLE ROW LEVEL SECURITY;
CREATE POLICY IF NOT EXISTS "public_rw" ON businesses FOR ALL USING (true) WITH CHECK (true);

SELECT 'businesses-Tabelle bereit: ' || count(*) || ' Datensätze' FROM businesses;

Korrekte businesses-Tabellenstruktur

          
CREATE TABLE IF NOT EXISTS businesses (
  id              BIGSERIAL PRIMARY KEY,
  name            TEXT,
  address         TEXT,
  phone           TEXT,
  website         TEXT,
  category        TEXT,
  rating          NUMERIC(3,1),
  opening_status  TEXT,
  maps_url        TEXT,
  lead_status     TEXT DEFAULT 'Open',
  assigned_agent  TEXT DEFAULT 'UNASSIGNED',
  notes           TEXT,
  last_called_at  TIMESTAMPTZ,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);
